rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Public
    match /public/{allPaths=**} {
      allow read;
      allow write: if false;
    }
  
    // DB Backups
    match /db_backups/{allPaths=**} {
      allow read: if request.auth != null && request.auth.token.authority == 5;
      allow write: if false;
    }

    // Candlestick Spreadsheets
    match /candlestick_spreadsheets/{allPaths=**} {
      allow read: if request.auth != null && request.auth.token.authority >= 2;
      allow write: if false;
    }

    // Arima Cache
    match /arima_cache/{allPaths=**} {
      allow read: if request.auth != null && request.auth.token.authority >= 3;
      allow write: if false;
    }
  }
}
